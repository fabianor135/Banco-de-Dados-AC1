    id_produto INT PRIMARY KEY,
    nome VARCHAR(255) NOT NULL,
    preço DECIMAL(10, 2) NOT NULL,
    estoque INT NOT NULL,
    perecível BOOLEAN NOT NULL,
    marca VARCHAR(100),
    nacionalidade VARCHAR(100)
);
